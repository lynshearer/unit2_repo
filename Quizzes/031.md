## Solution to Quiz #031

```.py
import requests
from matplotlib import pyplot as plt
import numpy as np

temp = []

req = requests.get('http://192.168.6.142/readings')
data = req.json()
readings = data["readings"][0]
for r in readings:
    print(r)
    if r['sensor_id']==1:
        temp.append(r['value'])

temp_610_800 = temp[610:801]
x_610_800 = []
for i in range (len(temp_610_800)):
    x_610_800.append(i)


# linear model
m, b = np.polyfit(x_610_800, temp_610_800, 1)
y_linear = []
x_linear = [0, x_610_800[-1]]
for i in x_linear: # [-1] means the last value
    y_linear.append(m*i + b)

# quadratic model
p0, p1, p2 = np.polyfit(x_610_800, temp_610_800, 2)
y_quad = []
for i in x_610_800:
    y_quad.append(p0*(i**2) + p1*i + p2)

plt.style.use('seaborn-whitegrid')
plt.plot(x_linear, y_linear, color='black')
plt.plot(x_610_800, y_quad, color='navy')
plt.scatter(x_610_800, temp_610_800)
plt.show()
```
